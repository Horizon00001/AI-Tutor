# 题目收藏功能 - 详细实现计划

## 功能概述

为教师提供个人教学资源库，支持收藏优质题目、分类管理、添加教学笔记。

---

## 功能模块设计

### 1. 数据库设计

```sql
-- 收藏文件夹表（支持嵌套）
CREATE TABLE teacher_folders (
    folder_id TEXT PRIMARY KEY,
    teacher_id TEXT NOT NULL,
    name TEXT NOT NULL,
    parent_id TEXT,  -- 父文件夹ID，NULL表示根文件夹
    color TEXT DEFAULT '#1890ff',  -- 文件夹颜色
    sort_order INTEGER DEFAULT 0,  -- 排序
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (parent_id) REFERENCES teacher_folders(folder_id),
    FOREIGN KEY (teacher_id) REFERENCES users(user_id)
);

-- 题目收藏表
CREATE TABLE teacher_collections (
    collection_id TEXT PRIMARY KEY,
    teacher_id TEXT NOT NULL,
    question_id TEXT NOT NULL,  -- 关联questions表
    folder_id TEXT,  -- 所属文件夹
    
    -- 题目信息（快照，防止原题被修改）
    title TEXT,
    content TEXT,
    answer TEXT,
    analysis TEXT,
    
    -- 教师标注
    tags TEXT,  -- JSON数组 ["标签1", "标签2"]
    difficulty_note TEXT,  -- 难度备注（如"学生易错"）
    teach_note TEXT,  -- 教学笔记
    common_errors TEXT,  -- 学生常见错误记录
    teach_points TEXT,  -- 讲解要点
    
    -- 使用统计
    use_count INTEGER DEFAULT 0,  -- 使用次数
    last_used_at TIMESTAMP,  -- 最后使用时间
    
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    
    FOREIGN KEY (teacher_id) REFERENCES users(user_id),
    FOREIGN KEY (question_id) REFERENCES questions(question_id),
    FOREIGN KEY (folder_id) REFERENCES teacher_folders(folder_id)
);

-- 收藏标签表（教师自定义标签）
CREATE TABLE collection_tags (
    tag_id TEXT PRIMARY KEY,
    teacher_id TEXT NOT NULL,
    tag_name TEXT NOT NULL,
    color TEXT DEFAULT '#52c41a',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (teacher_id) REFERENCES users(user_id),
    UNIQUE(teacher_id, tag_name)
);
```

### 2. API接口设计

#### 2.1 文件夹管理

```python
# 创建文件夹
POST /api/v1/collections/folders
{
    "name": "经典例题",
    "parent_id": null,  # null表示根目录
    "color": "#ff4d4f"
}

# 获取文件夹列表（树形结构）
GET /api/v1/collections/folders?teacher_id=xxx

# 更新文件夹
PUT /api/v1/collections/folders/{folder_id}
{
    "name": "新的名称",
    "color": "#1890ff"
}

# 删除文件夹（可设置是否移动子内容到根目录）
DELETE /api/v1/collections/folders/{folder_id}?move_to_root=true

# 移动文件夹
PUT /api/v1/collections/folders/{folder_id}/move
{
    "parent_id": "新的父文件夹ID"
}
```

#### 2.2 题目收藏

```python
# 收藏题目

GET /api/v1/collections/{collection_id}
```

### 5. 数据模型

```python
# api/models/schemas.py

class CreateFolderRequest(BaseModel):
    teacher_id: str
    name: str
    parent_id: Optional[str] = None
    color: Optional[str] = "#1890ff"

class FolderResponse(BaseModel):
    folder_id: str
    name: str
    parent_id: Optional[str]
    color: str
    children: Optional[List["FolderResponse"]] = None
    created_at: datetime

class CollectQuestionRequest(BaseModel):
    teacher_id: str
    question_id: str
    folder_id: Optional[str] = None
    tags: Optional[List[str]] = []
    difficulty_note: Optional[str] = None
    teach_note: Optional[str] = None
    common_errors: Optional[str] = None
    teach_points: Optional[str] = None

class CollectionResponse(BaseModel):
    collection_id: str
    question_id: str
    folder_id: Optional[str]
    folder_name: Optional[str]
    title: str
    content: str
    answer: Optional[str]
    analysis: Optional[str]
    tags: List[str]
    difficulty_note: Optional[str]
    teach_note: Optional[str]
    common_errors: Optional[str]
    teach_points: Optional[str]
    use_count: int
    last_used_at: Optional[datetime]
    created_at: datetime

class CreateTagRequest(BaseModel):
    teacher_id: str
    tag_name: str
    color: Optional[str] = "#52c41a"

class TagResponse(BaseModel):
    tag_id: str
    tag_name: str
    color: str
    use_count: int  # 使用次数
    created_at: datetime

class CollectionStatsResponse(BaseModel):
    total_count: int
    folder_count: int
    tag_count: int
    this_week_count: int
    most_used_tags: List[dict]
    recent_collections: List[dict]
```
